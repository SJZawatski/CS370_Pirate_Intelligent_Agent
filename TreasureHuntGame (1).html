This is later used by the agent for learning by experience, called "exploration". This notebook shows how to play a game. The pirate agent can move in four directions: left, right, up, and down.

While the agent primarily learns by experience through exploitation, often, the agent can choose to explore the environment to find previously undiscovered paths. This is called "exploration" and is defined by epsilon. This value is typically a lower value such as 0.1, which means for every ten attempts, the agent will attempt to learn by experience nine times and will randomly explore a new path one time. You are encouraged to try various values for the exploration factor and see how the algorithm performs.

def play_game(model, qmaze, pirate_cell):
    qmaze.reset(pirate_cell)
    envstate = qmaze.observe()
    while True:
        prev_envstate = envstate
        # get next action
        q = model.predict(prev_envstate)
        action = np.argmax(q[0])

        # apply action, get rewards and new state
        envstate, reward, game_status = qmaze.act(action)
        if game_status == 'win':
            return True
        elif game_status == 'lose':
            return False

def completion_check(model, qmaze):

def build_model(maze):
    model = Sequential()
    model.add(Dense(maze.size, input_shape=(maze.size,)))
    model.add(PReLU())
    model.add(Dense(maze.size))
    model.add(PReLU())
    model.add(Dense(num_actions))
    model.compile(optimizer='adam', loss='mse')
    return model

def qtrain(model, maze, **opt):
    """
    Deep Q-learning loop wired to the starter API.

    Works with:
      - TreasureMaze(maze): .reset(pirate_cell), .observe(), .act(action),
                            .is_solved(), .maze_size, .maze (np array)
      - GameExperience(model, max_memory): .remember([...]), .get_data(data_size)
      - model.predict(state) -> [q_values], model.fit(inputs, targets, epochs=1)
      - randomly_select_free_cell(qmaze) -> (r, c) free start cell
    """
    # ---------- hyperparams ----------
    global epsilon
    epsilon     = opt.get('epsilon', 1.0)                 # exploration
    # accept both 'epochs' and 'n_epoch'
    n_epoch     = opt.get('epochs', opt.get('n_epoch', 15000))
    max_memory  = opt.get('max_memory', 1000)             # replay size
    data_size   = opt.get('data_size', 50)                # minibatch size

    # ---------- env & replay ----------
    start_time  = datetime.datetime.now()
    qmaze       = TreasureMaze(maze)
    experience  = GameExperience(model, max_memory=max_memory)

    win_history = []                        # 1 if solved else 0
    hsize       = qmaze.maze.size // 2      # smoothing window
    win_rate    = 0.0

    # ---------- training loop ----------
    for epoch in range(n_epoch):
        # REQUIRED by your API: choose a free start cell for the pirate
        pirate_cell = randomly_select_free_cell(qmaze)
        qmaze.reset(pirate_cell)            # <-- fix: pass pirate start
        game_over = False

        # starting observation
        state = qmaze.observe()
        state = np.array(state, dtype=np.float32)
        if state.ndim == 1:
            state = state.reshape(1, -1)

        total_reward = 0.0

        while not game_over:
            # --- epsilon-greedy ---
            if np.random.rand() <= epsilon:
                action = np.random.randint(0, qmaze.maze_size)   # explore
            else:
                q_values = model.predict(state)[0]               # exploit
                action = int(np.argmax(q_values))

            # --- env step ---
            prev_state = state
            next_state, reward, game_over = qmaze.act(action)

            # shape next state for the model
            next_state = np.array(next_state, dtype=np.float32)
            if next_state.ndim == 1:
                next_state = next_state.reshape(1, -1)

            total_reward += reward

            # --- store transition & one SGD step ---
            experience.remember([prev_state, action, reward, next_state, game_over])
            inputs, targets = experience.get_data(data_size=data_size)
            model.fit(inputs, targets, epochs=1, verbose=0)

            state = next_state

        # decay epsilon but keep a small explore floor
        if epsilon > 0.10:
            epsilon *= 0.995

        # track win rate
        win_history.append(qmaze.is_solved())
        if len(win_history) > hsize:
            win_history = win_history[-hsize:]
        win_rate = float(sum(win_history)) / len(win_history)

        print(f"Epoch {epoch+1}/{n_epoch} | Win rate: {win_rate:.2f} | "
              f"Epsilon: {epsilon:.3f} | Reward: {total_reward:.2f}")

        # early stop if consistently solved
        if win_rate > 0.90:
            print(f"Reached 90% win rate at epoch {epoch+1}")
            break

    elapsed = datetime.datetime.now() - start_time
    print("Training finished in:", elapsed)
    return model, win_history

def randomly_select_free_cell(qmaze):
    """
    Return a random (row, col) that is free (value==0) in the underlying maze array.
    Your TreasureMaze exposes the numpy grid as qmaze.maze.
    """
    maze_arr = qmaze.maze                     # numpy array
    n_rows, n_cols = maze_arr.shape
    free_cells = [(r, c)
                  for r in range(n_rows)
                  for c in range(n_cols)
                  if maze_arr[r, c] == 0]     # 0 == free (as your comments note)
    return random.choice(free_cells)
